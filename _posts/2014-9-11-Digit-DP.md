### 数位dp

解决一类区间计数问题，一般模板为：
{% highlight c %}
int bit[N];
LL dfs(int pos,int s1,int s2,...,int pre,int limit){
    if(pos<0){
        ...// judge()
    }
    if(!limit && dp[pos][s1][s2][..][pre]!=-1){
        return dp[pos][s1][s2][..][pre];
    }
    int li=limit?bit[pos]:9;
    LL ans=0;
    for(int i=0;i<=li;i++){
        ans+=dfs(...,limit&&(i==li));
    }
    if(!limit){
        dp[pos][s1][s2][..][pre]=ans;
    }
    return ans;
}
LL solve(LL n){
    int cnt=0;
    while(n){
        bit[cnt++]=n%10;
        n/=10;
    }
    return dfs(pos-1,...,1);
}
